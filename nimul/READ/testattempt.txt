#include "MedicineAI.hpp"

int main() {
    // Access the medicine tracker
    MedicineAI::MedicineTracker tracker;
    tracker.LoadFromStorage("./medicine_data");
    
    // Generate resupply report
    int64_t target_date = std::time(nullptr) + (180 * 86400);  // 180 days
    auto report = tracker.GenerateResupplyReport(target_date, 180);
    
    std::cout << "Resupply Recommendations: " << report.recommendations.size() << std::endl;
    for (const auto& rec : report.recommendations) {
        std::cout << "  " << rec.medication_name << ": " 
                  << rec.recommended_amount << " doses" << std::endl;
    }
    
    return 0;
}
